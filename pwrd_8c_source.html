56 void Pwrd_sleep(void)
57 {
58  /* If the controls find it ok to go to sleep, door closed for a long time then set
59  * the WDT to 8 seconds. Enable the button interrupt to be able to wake it up from
60  * deep sleep.*/
61  if (sleepMode_E == PWRD_LONG_DEEP_SLEEP_E)
62  {
63  Wdtc_SetTimer(WDTC_8S_E, TRUE);
64  Butt_enableInterrupt();
65  }
66  else if (sleepMode_E == PWRD_INFINITE_SLEEP_E)
67  {
68  Wdtc_SetTimer(WDTC_OFF_E, FALSE);
69  Butt_enableInterrupt();
70  }
71  else
72  {
73  Wdtc_SetTimer(WDTC_16MS_E, TRUE);
74  }
75  /* Determine if timer shall be running in sleep */
76  if (sleepMode_E == PWRD_SLEEP_WITH_TIMER_RUNNING_E)
77  {
78  /* Sleep mode idle */
79  MCUCR = 0;
80  }
81  else
82  {
83  /* Sleep mode power-down */
84  MCUCR = _BV(SM1);
85  }
86 
87  cli(); /* Timed sequence make sure it's not disturbed */
88  MCUCR |= (_BV(BODS) | _BV(SE) | _BV(BODSE));
89  MCUCR &= ~_BV(BODSE);
90  sei(); /* To be able to wake-up */
91  asm volatile("sleep"::);
92 }
