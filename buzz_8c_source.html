43 inline void Buzz_init(void)
44 {
45 /* BuzzerFreq = F_CPU / (2 * Prescaler * (1 + OCR0A))
46  *
47  * In this case gives a buzzer frequency of 1690 Hz
48  */
49 #if F_CPU == 8000000
50  OCR0A = 36;
51 #elif F_CPU == 9600000
52  OCR0A = 43;
53 #else
54 #error "Correct compare value to CPU speed."
55 #endif
56 }

58 inline void Buzz_loop(void)
59 {
60  static tU08 counter_U08;
61 
62  if (soundType_E == BUZZ_ON_E)
63  {
64  turnOn();
65  counter_U08 = 0;
66  }
67  else if (soundType_E == BUZZ_ALARM_E)
68  {
69  if (++counter_U08 > BUZZ_ALARM_PERIOD_TIME / TICK)
70  {
71  counter_U08 = 0;
72  }
73  if (counter_U08 > BUZZ_ALARM_PERIOD_TIME / TICK / 2)
74  {
75  turnOn();
76  }
77  else
78  {
79  turnOff();
80  }
81  }
82  else
83  {
84  turnOff();
85  counter_U08 = 0;
86  }
87 }

89 static void turnOn(void)
90 {
91  /* Enable module before accessing registers */
92  PRR = PRR_INIT & ~_BV(PRTIM0);
93  /* Set to 'CTC' mode, toggle on match */
94  TCCR0A = _BV(COM0A0) | _BV(WGM01);
95  /* Start clock - prescaler 64 */
96  TCCR0B = _BV(CS01) | _BV(CS00);
97 }

99 static void turnOff(void)
100 {
101  /* Stop clock */
102  TCCR0B = 0;
103  /* Normal port operation, disconnect OC0A with potential of leaving
104  * pin state high. */
105  TCCR0A = 0;
106  /* Disconnect module */
107  PRR = PRR_INIT | _BV(PRTIM0);
108 }
