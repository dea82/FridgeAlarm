44 void Butt_Loop(void)
45 {
46  static tButt_State_str buttonRawOld_str;
47  tButt_State_E buttonRaw_E =
48  (GET_STATUS(BUTT_CFG) ? BUTT_RELEASED_E : BUTT_PRESSED_E);
49 
50  /* Is raw button glitching */
51  if (buttonRawOld_str.state_E == buttonRaw_E)
52  {
53  INC_U08(buttonRawOld_str.tickInState_U08);
54  }
55  else
56  {
57  buttonRawOld_str.tickInState_U08 = 0;
58  }
59 
60  /* Check if raw value is stable */
61  if ((buttonRawOld_str.tickInState_U08 >= FILTER_TIME / TICK) &&
62  (buttState_str.state_E != buttonRaw_E))
63  {
64  buttState_str.state_E = buttonRaw_E;
65  buttState_str.tickInState_U08 = 0;
66  }
67  else
68  {
69  INC_U08(buttState_str.tickInState_U08);
70  }
71 
72  buttonRawOld_str.state_E = buttonRaw_E;
73 }
