64 {
65  /* ATTENTION! Initialization code can be found in boot.S */
66 
67  /* Check for WDT reset (save code size and assume true) - if a runaway pointer enables it,
68  * then it must be disabled here because it's kept after a reset! Ref. AVR132 chap 2.4.
69  * Reset status register because it will cause a new WDT reset if WDRF is set. Make sure
70  * to save it for later useage to detect if abnormal reset has occured.
71  */
72  tU08 statusRegister_U08 = MCUSR;
73  MCUSR = 0;
74  WDTCR = _BV(WDCE) | _BV(WDE);
75  WDTCR = 0;
76 
77

81  /* Initialize I/O Ports */

84

87  Dsen_init();

89  Cont_init();

98  /* Enable global interrupt otherwise dsen will not be able to wake up

101 
109  if(!(statusRegister_U08 & _BV(PORF)))
110  {
111  Butt_enableInterrupt();
