85  /* Check for WDT reset (save code size and assume true) - if a runaway pointer enables it,
86  * then it must be disabled here because it's kept after a reset! Ref. AVR132 chap 2.4.
87  * Reset status register because it will cause a new WDT reset if WDRF is set. Make sure
88  * to save it for later useage to detect if abnormal reset has occured.
89  */
90  {

92  MCUSR = 0;
93  WDTCR = _BV(WDCE) | _BV(WDE);
94  WDTCR = 0;

97  ACSR |= _BV(ACD);
98 
99  /* Initialize I/O Ports */
100  DDRB = DDRB_INIT;
101  PORTB |= PORTB_INIT; /* Optimization, only one bit is changed */
102

104  Butt_Init();

107  Cont_Init();

109  Buzz_Init();
110  Ledc_Init();

112  /* Enable global interrupt otherwise door will not be able to wake up
113  * from ADC Noise reduction mode or infinite sleep mode. */
114  asm volatile("sei"::);
115 
123  if(!(statusRegister_U08 & _BV(PORF)))
124  {
