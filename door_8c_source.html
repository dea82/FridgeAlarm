57 inline void Door_Init(void)
58 {
59 #if F_CPU == 9600000
60  /* Prescaler for ADC clock is set to 64, which gives a ADC clock of 150 kHz. */
61  ADCSRA = (1 << ADPS2) | (1 << ADPS1);
62 #elif F_CPU == 1000000
63  /* Prescaler for ADC clock is set to 8, which gives a ADC clock of 125 kHz. */
64  ADCSRA = (1 << ADPS1) | (1 << ADPS0);
65 #elif F_CPU == 1200000
66  /* Prescaler for ADC clock is set to 8, which gives a ADC clock of 150 kHz. */
67  ADCSRA = (1 << ADPS1) | (1 << ADPS0);
68 #elif F_CPU == 4800000
69  /* Prescaler for ADC clock is set to 32, which gives a ADC clock of 150 kHz. */
70  ADCSRA = (1 << ADPS2) | (1 << ADPS0);
71 #elif F_CPU == 600000
72  /* Prescaler for ADC clock is set to 4, which gives a ADC clock of 150 kHz. */
73  ADCSRA = (1 << ADPS1);
74 #elif F_CPU == 8000000
75  /* Prescaler for ADC clock is set to 64, which gives a ADC clock of 125 kHz. */
76  ADCSRA = (1 << ADPS2) | (1 << ADPS1);
77 #else
78 #error "Prescaler for ADC conversion is not supported for this CPU clock."
79 #endif
80 
81  /* Connect pin to ADC with 10-bit precision */
82  ADMUX = (_BV(MUX1) | _BV(MUX0));
83 
84  doorClosed_U16 = (Eepr_Read_U08(CALIBRATED_CLOSED_POS_ADDRESS) << 8);
85  doorClosed_U16 += Eepr_Read_U08(CALIBRATED_CLOSED_POS_ADDRESS + 1);
86 }

88 inline void Door_Loop(void)
89 {
90  doorPos_U16 = getDoorRawPos_U16();
91 
92  tDoor_Position_E newposition_E;
93  /* Algorithm handles both direction of magnet field */
94  if (doorClosed_U16 < UNAFFECTED_SENSOR_VALUE)
95  {
96  if (doorPos_U16 > doorClosed_U16 + DOOR_CLOSED_OFFSET)
97  {
98  newposition_E = DOOR_OPEN_E;
99  }
100  else
101  {
102  newposition_E = DOOR_CLOSED_E;
103  }
104  }
105  else
106  {
107  if (doorPos_U16 < doorClosed_U16 - DOOR_CLOSED_OFFSET)
108  {
109  newposition_E = DOOR_OPEN_E;
110  }
111  else
112  {
113  newposition_E = DOOR_CLOSED_E;
114  }
115  }
116 
117  if (doorState_str.position_E == newposition_E)
118  {
119  INC_U16(doorState_str.ticksInState_U16);
120  }
121  else
122  {
123  doorState_str.ticksInState_U16 = 0;
124  }
125  doorState_str.position_E = newposition_E;
126 
127 }

133 static tU16 getDoorRawPos_U16(void)
134 {
135  tU16 sensorValue_U16;
136 
137 
138  /* Prepare for sleep - check if timer is running.
139  * Assume that timer is not running if module is disconnected. */
140  if (PRR & _BV(PRTIM0))
141  {
142  /* Sleep mode ADC noise reduction */
143  MCUCR = _BV(SE) | _BV(SM0);
144  }
145  else
146  {
147  /* Sleep mode idle */
148  MCUCR = _BV(SE);
149  }
150 
151  /* Turn on sensor */
152  IO_SET(DOOR_SWITCH_CFG);
153 
154  /* Activate ADC module */
155  PRR &= ~_BV(PRADC);
156 
157  /* According to (Allegro 1302) data sheet of sensor star-up time is 5 us.*/
158  _delay_us(5);
159 
160  /* Clear any pending interrupt - should not happen */
161  ADCSRA |= _BV(ADIF);
162  /* Start conversion */
163  ADCSRA |= _BV(ADEN) | _BV(ADSC) | _BV(ADIE);
164 
165  /* Necessary check if any other interrupt wakes the MCU. */
166  while (ADCSRA & _BV(ADSC))
167  {
168  /* Wait for conversion */
169  asm volatile("sleep"::);
170  }
171 
172  /* Disable interrupt and ADC. */
173  ADCSRA &= ~(_BV(ADEN) | _BV(ADIE));
174 
175  /* Turn off sensor */
176  IO_CLR(DOOR_SWITCH_CFG);
177 
178  /* Power savings */
179  PRR |= _BV(PRADC);
180 
181  sensorValue_U16 = ADCL;
182  sensorValue_U16 |= ((ADCH & 0x3) << 8);
183 
184  return sensorValue_U16;
185 }
