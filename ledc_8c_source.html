58 {
59  static tU08 counter_U08;
60  if (++counter_U08 > blinkPeriodTime_U08c / tick_U08c)
61  {
62  counter_U08 = 0;
63  }
64  switch (ledState_E)
65  {
66  case LEDC_OFF_E:
67  IO_CLR(GREEN_LED_CFG);
68  IO_CLR(RED_LED_CFG);
69  break;
70  case LEDC_GREEN_E:
71  setGpioPinHi(greenLed.anodePin);
72  setGpioPinLo(redLed.anodePin);
73  break;
74  case LEDC_RED_E:
75  IO_CLR(GREEN_LED_CFG);
76  IO_SET(RED_LED_CFG);

78 
79  case LEDC_ORANGE_E:
80  IO_SET(GREEN_LED_CFG);
81  IO_SET(RED_LED_CFG);
82  break;
83  case LEDC_GREEN_BLINK_E:
84  if (counter_U08 > (blinkPeriodTime_U08c / tick_U08c / HALF_DIVISOR))
85  {
86  IO_SET(GREEN_LED_CFG);

88  else

92  IO_CLR(RED_LED_CFG);

96  {

99  else

101  IO_CLR(RED_LED_CFG);
102  }
