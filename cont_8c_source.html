53 void Cont_Loop(void)
54 {
55  tCalibrationState_E calibrationState_E = getCalibrationState_E();
56 
57  /* Sets it default. */
58  Pwrd_SetSleepMode(PWRD_SHORT_DEEP_SLEEP_E);
59 
60  if (calibrationState_E == CALIBRATION_SUCCESS_E)
61  {
62  Buzz_SetSound(BUZZ_OFF_E);
63  Ledc_SetState(LEDC_GREEN_BLINK_E);
64  }
65  else if (calibrationState_E == CALIBRATION_FAIL_E)
66  {
67  Buzz_SetSound(BUZZ_ON_E);
68  Ledc_SetState(LEDC_RED_BLINK_E);
69  Pwrd_SetSleepMode(PWRD_SLEEP_WITH_TIMER_RUNNING_E);
70  }
71  else
72  {
73  normalModeStateMachine();
74  }
75 }

77 static tCalibrationState_E getCalibrationState_E(void)
78 {
79  static tCalibrationState_E calibrationState_E;
80  tButt_State_str buttState_str = Butt_GetState_str();
81 
82  /* Calibration state */
83  if ((buttState_str.state_E == BUTT_PRESSED_E) &&
84  (buttState_str.tickInState_U08 * tick_U08c > calibrationTimeButton_U16c) &&
85  (calibrationState_E == NO_CALIBRATION_E))
86  {
87  /* Enters calibration mode */
88  if (Door_StoreClosedPos_B() == TRUE)
89  {
90  calibrationState_E = CALIBRATION_SUCCESS_E;
91  }
92  else
93  {
94  calibrationState_E = CALIBRATION_FAIL_E;
95  }
96  }
97  else
98  {
99  /* Leaves calibration mode */
100  if (buttState_str.state_E == BUTT_RELEASED_E)
101  {
102  calibrationState_E = NO_CALIBRATION_E;
103  }
104  }
105 
106  return calibrationState_E;
107 }

109 static void normalModeStateMachine(void)
110 {
111  static tB inhibitAlarm_B;
112  static tU08 counter_U08;
113  tButt_State_str buttState_str = Butt_GetState_str();
114  tDoor_State_str doorState_str = Door_GetState_str();
115 
116  if (doorState_str.position_E == DOOR_CLOSED_E)
117  {
118  Buzz_SetSound(BUZZ_OFF_E);
119  if ((doorState_str.ticksInState_U16 > lightsOnDoorClosed_U16c / tick_U08c) &&
120  (counter_U08 > minTimeAwake_U08c / tick_U08c) &&
121  (buttState_str.state_E == BUTT_RELEASED_E))
122  {
123  /* When door has been closed for a long time and no activity
124  * on button it's OK to go to deep sleep. */
125  counter_U08 = 0;
126  Ledc_SetState(LEDC_OFF_E);
127  Pwrd_SetSleepMode(PWRD_LONG_DEEP_SLEEP_E);
128  }
129  else
130  {
131  /* Door has not been closed long enough to go to deep sleep. */
132  Util_safeIncrementU08(&counter_U08);
133  Ledc_SetState(LEDC_GREEN_E);
134  }
135  /* Reset */
136  inhibitAlarm_B = FALSE;
137  }
138  else /* DOOR_OPEN_E */
139  {
140  counter_U08 = 0;
141 
142  if (doorState_str.ticksInState_U16 > alarmOpen_U16c / tick_U08c)
143  {
144  /* Door has been open too long, sound the alarm if button is not
145  * inhibiting it from going off. */
146  if (buttState_str.state_E == BUTT_PRESSED_E)
147  {
148  inhibitAlarm_B = TRUE;
149  }
150  if (inhibitAlarm_B == TRUE)
151  {
152  Buzz_SetSound(BUZZ_OFF_E);
153  }
154  else
155  {
156  Buzz_SetSound(BUZZ_ALARM_E);
157  /* Necessary to have the timer running due to the buzzer. */
158  Pwrd_SetSleepMode(PWRD_SLEEP_WITH_TIMER_RUNNING_E);
159  }
160  Ledc_SetState(LEDC_RED_BLINK_E);
161  }
162  else
163  {
164  /* Door open but not long enough to sound the alarm. */
165  Ledc_SetState(LEDC_RED_E);
166  Buzz_SetSound(BUZZ_OFF_E);
167  }
168  }
