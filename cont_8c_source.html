51 inline void Cont_loop(void)
52 {
53  tCalibrationState_E calibrationState_E = getCalibrationState_E();
54 
55  /* Sets it default. */
56  Pwrd_setSleepMode(PWRD_SHORT_DEEP_SLEEP_E);
57 
58  if (calibrationState_E == CALIBRATION_SUCCESS_E)
59  {
60  Buzz_setSound(BUZZ_OFF_E);
61  Ledc_setState(LEDC_GREEN_BLINK_E);
62  }
63  else if (calibrationState_E == CALIBRATION_FAIL_E)
64  {
65  Buzz_setSound(BUZZ_ON_E);
66  Ledc_setState(LEDC_RED_BLINK_E);
67  Pwrd_setSleepMode(PWRD_SLEEP_WITH_TIMER_RUNNING_E);
68  }
69  else
70  {
71  normalModeStateMachine();
72  }
73 }

75 static tCalibrationState_E getCalibrationState_E(void)
76 {
77  static tCalibrationState_E calibrationState_E;
78  tButt_State_str buttState_str = Butt_getState_str();
79 
80  /* Calibration state */
81  if ((buttState_str.state_E == BUTT_PRESSED_E) &&
82  (buttState_str.tickInState_U08 * TICK > CALIBRATION_TIME_BUTTON) &&
83  (calibrationState_E == NO_CALIBRATION_E))
84  {
85  /* Enters calibration mode */
86  if (Dsen_storeClosedPos_B() == TRUE)
87  {
88  calibrationState_E = CALIBRATION_SUCCESS_E;
89  }
90  else
91  {
92  calibrationState_E = CALIBRATION_FAIL_E;
93  }
94  }
95  else
96  {
97  /* Leaves calibration mode */
98  if (buttState_str.state_E == BUTT_RELEASED_E)
99  {
100  calibrationState_E = NO_CALIBRATION_E;
101  }
102  }
103 
104  return calibrationState_E;
105 }

107 static void normalModeStateMachine(void)
108 {
109  static tB inhibitAlarm_B;
110  static tU08 counter_U08;
111  tButt_State_str buttState_str = Butt_getState_str();
112  tDsen_doorState_str doorState_str = Dsen_getDoorState_str();
113 
114  if (doorState_str.doorState_E == DSEN_CLOSED_E)
115  {
116  Buzz_setSound(BUZZ_OFF_E);
117  if ((doorState_str.ticksInState_U16 > LIGHTS_ON_DOOR_CLOSED / TICK) &&
118  (counter_U08 > MIN_TIME_AWAKE / TICK) &&
119  (buttState_str.state_E == BUTT_RELEASED_E))
120  {
121  /* When door has been closed for a long time and no activity
122  * on button it's OK to go to deep sleep. */
123  counter_U08 = 0;
124  Ledc_setState(LEDC_OFF_E);
125  Pwrd_setSleepMode(PWRD_LONG_DEEP_SLEEP_E);
126  }
127  else
128  {
129  /* Door has not been closed long enough to go to deep sleep. */
130  INC_U08(counter_U08);
131  Ledc_setState(LEDC_GREEN_E);
132  }
133  /* Reset */
134  inhibitAlarm_B = FALSE;
135  }
136  else /* DSEN_OPEN_E */
137  {
138  counter_U08 = 0;
139 
140  if (doorState_str.ticksInState_U16 > ALARM_OPEN / TICK)
141  {
142  /* Door has been open too long, sound the alarm if button is not
143  * inhibiting it from going off. */
144  if (buttState_str.state_E == BUTT_PRESSED_E)
145  {
146  inhibitAlarm_B = TRUE;
147  }
148  if (inhibitAlarm_B == TRUE)
149  {
150  Buzz_setSound(BUZZ_OFF_E);
151  }
152  else
153  {
154  Buzz_setSound(BUZZ_ALARM_E);
155  /* Necessary to have the timer running due to the buzzer. */
156  Pwrd_setSleepMode(PWRD_SLEEP_WITH_TIMER_RUNNING_E);
157  }
158  Ledc_setState(LEDC_RED_BLINK_E);
159  }
160  else
161  {
162  /* Door open but not long enough to sound the alarm. */
163  Ledc_setState(LEDC_RED_E);
164  Buzz_setSound(BUZZ_OFF_E);
165  }
166  }
167 }
